  6. Escribe una función que genere los n primeros números de la secuencia de Fibonacci

    function generarFibonacci(n) {
    const secuencia = [];

    for (let i = 0; i < n; i++) {
    if (i === 0) {
    secuencia.push(0);
    } else if (i === 1) {
    secuencia.push(1);
    } else {
    secuencia.push(secuencia[i - 1] + secuencia[i - 2]);
    }
    }
    return secuencia;
    }

    console.log(generarFibonacci(10));
