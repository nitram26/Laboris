  7. Crea un precio base y un porcentaje de descuento, luego usa una propiedad computada para calcular el precio final y mostrarlo en el template. 
    <template>
<div class="bg-teal-lighten-2 text-white text-center py-5">
  <h1>ProbatioVue</h1>
  <h2>La compra es de: $ {{ compra }}.00</h2>
  <h2>Su IGV 18% es: {{ igv }}.00</h2> <br>
  <h2>El pago mas IGV es: $ {{ precio }}.00</h2>
</div>
</template>
<!-- +++++++++++++++++++++++++++++++++++++ -->
<script setup>
import { computed, ref, } from 'vue';

let compra = 2500;
let igv = `${compra * 0.18}`;
let precio = ref(computed(() => {
  return compra * 1.18;
}));

</script>
