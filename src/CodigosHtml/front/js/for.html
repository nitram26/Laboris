function contarVocales(frase) {	
  // Expresión regular para buscar todas las vocales (incluye mayúsculas)	
  const vocales = frase.match(/[aeiouáéíóúAEIOUÁÉÍÓÚ]/g);	
	
  // Si no hay vocales, retorna 0	
  return vocales ? vocales.length : 0;	
}

function contarVocales(frase) {	
  // Definimos las vocales	
  let vocales = "aeiouáéíóúAEIOUÁÉÍÓÚ";	
  let contador = 0;	
	
  // Recorremos cada carácter de la frase	
  for (let letra of frase) {	
    if (vocales.includes(letra)) {	
      contador++;	
    }	
  }	
	
  return contador;	
}	
	
let frasse = "Hola, soy Cristian y me gusta el JavaScript"; 
console.log(contarVocales(frasse));
