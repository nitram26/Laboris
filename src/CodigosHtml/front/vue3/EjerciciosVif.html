    mayor a 600 => Cliente en Proceso
    de lo contrario => En Inicio

    pero ambas funcionan en cualquier caso

    <div class="bg-teal-lighten-2 text-white text-center py-5">
      <h1>ProbatioVue</h1>
      <h2 v-if="ahorro > 1000" class="uno">Cliente en Progreso</h2>
      <h2 v-else-if="ahorro > 500" class="dos">Cliente en Espera</h2>
      <h2 v-else-if="ahorro > 0" class="tres">Cliente en Inicio</h2>

      <v-text-field label="Ingresa el monto de Ahorro" v-model="ahorro" />
    </div>
  </template>
